CREATE PROCEDURE [dbo].[UpdateImageDetails] (
	@Ref varchar(10),
	@Title varchar(255),
	@Caption text,
	@SectionId int        
)
AS
UPDATE [Images] SET 
	[Title] = @Title, 
	[Caption] = @Caption,
	[SectionId] = @SectionId
WHERE [Ref] = @Ref


GO
