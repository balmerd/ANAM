CREATE PROCEDURE [dbo].[GetAllSectionText]
AS
SELECT s.Id, s.Name, st.SectionText 
FROM SectionText st INNER JOIN Sections s ON s.Id = st.SectionId 
WHERE s.Name <> 'Unassigned'
ORDER BY s.Name



GO
